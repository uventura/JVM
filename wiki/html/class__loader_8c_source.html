50 void class_loader_resolve(ClassFile *class_file, char *class_path)
51 {
52  cp_info *cp;
53  for (cp = class_file->constant_pool; cp < class_file->constant_pool + class_file->constant_pool_count; cp++)
54  {
55  if (cp->tag == CONSTANT_Class)
56  {
57  u2 index = cp->info.Class.name_index - 1;
58  u2 size = class_file->constant_pool[index].info.Utf8.length;
59 
60  char class_name[size + 1];
61  get_utf8_value(index, class_file->constant_pool, class_name);
62 
63  // Check if is Java Class

68  if (!strcmp(prefix, "java/"))
69  continue;
70  }
