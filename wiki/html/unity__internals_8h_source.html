91/* Determine the size of an int, if not already specified.
92 * We cannot use sizeof(int), because it is not yet defined
93 * at this stage in the translation of the C program.
94 * Also sizeof(int) does return the size in addressable units on all platforms,
95 * which may not necessarily be the size in bytes.
96 * Therefore, infer it from UINT_MAX if possible. */
97#ifndef UNITY_INT_WIDTH
98#ifdef UINT_MAX
99#if (UINT_MAX == 0xFFFF)
100#define UNITY_INT_WIDTH (16)
101#elif (UINT_MAX == 0xFFFFFFFF)
102#define UNITY_INT_WIDTH (32)
103#elif (UINT_MAX == 0xFFFFFFFFFFFFFFFF)
104#define UNITY_INT_WIDTH (64)
105#endif
106#else /* Set to default */
107#define UNITY_INT_WIDTH (32)
108#endif /* UINT_MAX */
109#endif
110
111/* Determine the size of a long, if not already specified. */
112#ifndef UNITY_LONG_WIDTH
113#ifdef ULONG_MAX
114#if (ULONG_MAX == 0xFFFF)
115#define UNITY_LONG_WIDTH (16)
116#elif (ULONG_MAX == 0xFFFFFFFF)
117#define UNITY_LONG_WIDTH (32)
118#elif (ULONG_MAX == 0xFFFFFFFFFFFFFFFF)
119#define UNITY_LONG_WIDTH (64)
120#endif
121#else /* Set to default */
122#define UNITY_LONG_WIDTH (32)
123#endif /* ULONG_MAX */
124#endif
125
126/* Determine the size of a pointer, if not already specified. */
127#ifndef UNITY_POINTER_WIDTH
128#ifdef UINTPTR_MAX
129#if (UINTPTR_MAX <= 0xFFFF)
130#define UNITY_POINTER_WIDTH (16)
131#elif (UINTPTR_MAX <= 0xFFFFFFFF)
132#define UNITY_POINTER_WIDTH (32)
133#elif (UINTPTR_MAX <= 0xFFFFFFFFFFFFFFFF)
134#define UNITY_POINTER_WIDTH (64)
135#endif
136#else /* Set to default */
137#define UNITY_POINTER_WIDTH UNITY_LONG_WIDTH
138#endif /* UINTPTR_MAX */
139#endif
